# 范例

const vm = new Vue({
    template: '<div>{{ b }}</div>',
    data: {
        a: 1
    },
    computed: {
        b() {
            return this.a + 1
        }
    }
}).$mount()

# 计算属性初始化

src/core/instance/state.js

export function initState (vm) {
  vm._watchers = []
  const opts = vm.$options
  if (opts.props) initProps(vm, opts.props)
  if (opts.methods) initMethods(vm, opts.methods)
  if (opts.data) {
    initData(vm)
  }
  // 初始化计算属性
  if (opts.computed) initComputed(vm, opts.computed)
}

初始化计算属性相关代码

// 该函数的主要作用：创建一个 computed watcher，并保存到 vm._computedWatchers。
function initComputed(vm, computed) {
    const watchers = vm._computedWatchers = Object.create(null)
    for (const key in computed) {
        const userDef = computed[key]
        const getter = typeof userDef === 'function' ? userDef : userDef.get

        // create internal watcher for the computed property.
        watchers[key] = new Watcher(
            vm,
            getter || noop,
            noop,
            { computed: true }
        );

        // 删除了一些判断，理解主要逻辑
        defineComputed(vm, key, userDef)
    }
}

export function defineComputed(target, key, userDef) {
    // 只考虑 userDef 为 function 的情况
    // 删除了一些代码
    sharedPropertyDefinition.get = createComputedGetter(key)
    sharedPropertyDefinition.set = noop
    Object.defineProperty(target, key, sharedPropertyDefinition)
}

function createComputedGetter(key) {
    return function computedGetter() {
        const watcher = this._computedWatchers && this._computedWatchers[key]
        if (watcher) {
            watcher.depend()
            return watcher.evaluate()
        }
    }
}

上面的代码总结起来主要完成了两项工作：

  1. 为配置中的每一个计算属性生成一个 computed watcher, 这些 computed watcher 保存到了 vm._computedWatchers 中。
  2. 使用Object.defineProperty,把计算属性定义到vm上。

通过上面的分析，我们知道计算属性最终会被定义到vm上（通过 Object.defineProperty）。computedGetter 就是定义中的get函数。接下来我们重点关注一下该函数。

# computedGetter

computedGetter 函数，分三部分理解

  1. 从 _computedWatchers 中找到计算属性 key 对应的 watcher
  2. watcher.depend()
  3. watcher.evaluate()

# watcher.depend()

export default class Watcher {
  constructor (
    vm: Component,
    expOrFn: string | Function,
    cb: Function,
    options?: ?Object,
    isRenderWatcher?: boolean
  ) {
    // computed watcher实例化时没有调用 get 方法
    // computed watcher内部增加了一个dep 对象
    if (this.computed) {
      this.value = undefined
      this.dep = new Dep()
    } else {
      this.value = this.get()
    }
  }

  evaluate () {
    if (this.dirty) {
      this.value = this.get()
      this.dirty = false
    }
    return this.value
  }
	
  // computed watcher 特有的方法
  depend () {
    if (this.dep && Dep.target) {
      this.dep.depend()
    }
  }

}

computed 属性特有的逻辑

  1. computed watcher 特有的方法depend
  2. computed watcher 有一个依赖收集器 dep
  3. computed watcher 实例化的时候没有调用this.get()直接求值

知道了上面3点后，我们再看一下depend具体干什么用的？

我们先把相关的代码整理一下

// watcher.js
depend () {
    if (this.dep && Dep.target) {
      this.dep.depend()
    }
}

// dep.js
depend () {
    if (Dep.target) {
      Dep.target.addDep(this)
    }
}

// watcher.js
addDep (dep) {
    const id = dep.id
    if (!this.newDepIds.has(id)) {
      this.newDepIds.add(id)
      this.newDeps.push(dep)
      if (!this.depIds.has(id)) {
        // 主要关注
        dep.addSub(this)
      }
    }
}

// dep.js
addSub (sub: Watcher) {
   this.subs.push(sub)
}

上面的代码完成了什么功能呢？

// watcher.js
depend () {
    if (this.dep && Dep.target) {
      // 把 Dep.target 存储到 this.dep.subs 中
      this.dep.depend()
    }
}

Dep.target 是什么呢？

Dep.target 保存的是当前正在求值的 watcher。可以从下面的代码看出

// watcher.js
get () {
    // 设置 Dep.target = this
    pushTarget(this)
    try {
      // 求值
      value = this.getter.call(vm, vm)
    } catch (e) {

    } finally {
      popTarget()
    }
    return value
}

通俗的说，computed watcher 中的 dep 收集的是依赖于该计算属性的 watcher。

# watcher.evaluate()

evaluate作用就是求值

// watcher.js
evaluate () {
    if (this.dirty) {
      this.value = this.get()
      this.dirty = false
    }
    return this.value
}

get () {
    // 设置 Dep.target = this
    pushTarget(this)
    try {
      // 求值
      value = this.getter.call(vm, vm)
    } catch (e) {

    } finally {
      popTarget()
    }
    return value
}

通过范例我们知道计算属性 b 等于 a + 1 (计算属性b对应的watcher，我们记做b-watcher)。对计算属性 b 求值的时候，会执行 get 方法，该方法首先设置 Dep.target 为 b-watcher。然后执行 this.getter 的时候，会触发 a 属性的get函数。在该函数内部会收集b-watcher到a属性对应的依赖收集器dep。这样我们改变 a 的时候，就可以执行 dep.notify() 通知到。相关代码如下：

src/core/observer/index.js

Object.defineProperty(obj, key, {
    enumerable: true,
    configurable: true,
    get: function reactiveGetter() {
        const value = getter ? getter.call(obj) : val
        if (Dep.target) {
            dep.depend()
        }
        return value
    },
    set: function reactiveSetter(newVal) {
        dep.notify()
    }
})

# 触发更新

当我们执行 vm.a = 2 的时候，会开始更新过程 dep.notify()

// dep.js
notify () {
    /*
      this.subs.slice() 对数组做拷贝，对订阅者列表做稳定化处理
    */
    // stabilize the subscriber list first
    const subs = this.subs.slice()
    for (let i = 0, l = subs.length; i < l; i++) {
      this.subs[i].update()
    }
}

具体的更新逻辑如下：

// watcher.js
update() {
    if (this.computed) {
        this.getAndInvoke(() => {
            this.dep.notify()
        })
    }
}

getAndInvoke(cb) {
    const value = this.get()
    // 只有值改变以后才触发
    if (value !== this.value || isObject(value) || this.deep) {
        const oldValue = this.value
        this.value = value
        this.dirty = false
        cb.call(this.vm, value, oldValue)
    }
}

这里主要关注一下 update 中的this.dep.notify()

通知的就是依赖于该计算属性的watcher。
